\section{Phần 1: Phép khử Gauss và Các Ứng Dụng}

\subsection{Cơ sở lý thuyết}
\subsubsection{Hệ phương trình tuyến tính và Ma trận tăng cường}
Xét hệ phương trình tuyến tính có dạng tổng quát:
\begin{equation}
    Ax = b
\end{equation}
Trong đó:
\begin{itemize}
    \item $A \in \mathbb{R}^{m \times n}$: Ma trận hệ số.
    \item $x \in \mathbb{R}^n$: Vector ẩn số.
    \item $b \in \mathbb{R}^m$: Vector vế phải.
\end{itemize}
Để giải hệ phương trình, ta thiết lập Ma trận tăng cường $[A|b]$ bằng cách ghép vector $b$ vào bên phải ma trận $A$.

\subsubsection{Các phép biến đổi dòng sơ cấp}
Thuật toán dựa trên ba phép biến đổi cơ bản không làm thay đổi tập nghiệm của hệ phương trình:
\begin{enumerate}
    \item Hoán đổi dòng: $R_i \leftrightarrow R_j$
    \item Nhân dòng với một hằng số khác 0: $R_i \leftarrow c \cdot R_i \quad (c \neq 0)$
    \item Cộng vào một dòng bội số của dòng khác: $R_i \leftarrow R_i + c \cdot R_j$
\end{enumerate}

\subsubsection{Dạng bậc thang (REF) và Bậc thang rút gọn (RREF)}
\begin{itemize}
    \item \textbf{Dạng bậc thang (Row Echelon Form - REF):} Các dòng bằng 0 nằm dưới cùng; phần tử khác 0 đầu tiên của một dòng (phần tử chốt - pivot) luôn nằm bên phải phần tử chốt của dòng phía trên nó.
    \item \textbf{Dạng bậc thang rút gọn (Reduced Row Echelon Form - RREF):} Là dạng REF nhưng mọi phần tử chốt đều bằng $1$ và là phần tử khác 0 duy nhất trong cột chứa nó.
\end{itemize}

\subsubsection{Khử Gauss có chọn phần tử chốt (Partial Pivoting)}
Để tăng độ ổn định số học và tránh lỗi chia cho $0$, ta chọn phần tử chốt có giá trị tuyệt đối lớn nhất trong cột hiện tại trước khi thực hiện phép khử:
\begin{equation}
    |a_{pk}| = \max_{i \geq k} |a_{ik}|
\end{equation}

\subsubsection{Tính định thức qua khử Gauss}
Sau khi đưa ma trận $A$ về dạng tam giác trên $U$ bằng phép khử Gauss với $s$ lần hoán đổi dòng, định thức được tính theo công thức:
\begin{equation}
    \det(A) = (-1)^s \cdot \prod_{i=1}^{n} u_{ii}
\end{equation}

\subsubsection{Ma trận nghịch đảo bằng Gauss-Jordan}
\textbf{Định lý:} Ma trận $A \in \mathbb{R}^{n \times n}$ khả nghịch khi và chỉ khi $\det(A) \neq 0$.

Ma trận nghịch đảo $A^{-1}$ được tìm bằng cách thực hiện các phép biến đổi dòng sơ cấp đồng thời trên ma trận ghép $[A | I_n]$ cho đến khi ma trận bên trái đạt dạng đơn vị:
\begin{equation}
    [A | I_n] \xrightarrow{\text{biến đổi dòng}} [I_n | A^{-1}]
\end{equation}

\subsubsection{Hạng ma trận và cơ sở}
\textbf{Định nghĩa:} Hạng của ma trận $A$, ký hiệu là $\text{rank}(A)$, bằng số dòng khác 0 trong dạng REF của $A$, hoặc tương đương, bằng số cột pivot. Từ dạng RREF, ta xác định được ba không gian quan trọng:
\begin{itemize}
    \item \textbf{Không gian cột $\mathcal{C}(A)$:} Sinh bởi các cột pivot của $A$ gốc.
    \item \textbf{Không gian dòng $\mathcal{R}(A)$:} Sinh bởi các dòng khác 0 trong RREF.
    \item \textbf{Không gian nghiệm $\mathcal{N}(A)$:} Tập nghiệm của hệ thuần nhất $Ax = 0$.
\end{itemize}

\subsection{Thuật toán khử Gauss (Partial Pivoting)}
Quy trình thực hiện được nhóm cài đặt gồm hai giai đoạn chính:
\begin{itemize}
    \item \textbf{Giai đoạn 1: Khử xuôi (Forward Elimination)}
    \begin{enumerate}
        \item Khởi tạo ma trận tăng cường $M = [A|b]$.
        \item Với mỗi cột $k$: 
        \item Tìm dòng $p$ chứa phần tử có trị tuyệt đối lớn nhất kể từ dòng $k$ trở xuống.
        \item Hoán đổi dòng $k$ và dòng $p$.
        \item Với mỗi dòng $i > k$: Thực hiện phép biến đổi dòng để triệt tiêu các phần tử bên dưới phần tử chốt $M_{kk}$.
    \end{enumerate}
    
    \item \textbf{Giai đoạn 2: Thế ngược (Back Substitution)}\\
    Tìm nghiệm $x_i$ theo công thức truy hồi từ dòng cuối cùng lên:
    \begin{equation}
        x_i = \frac{M_{i, n+1} - \sum_{j=i+1}^{n} M_{ij} \cdot x_j}{M_{ii}}
    \end{equation}
\end{itemize}

\subsection{Triển khai và cài đặt chi tiết (Phân tích mã nguồn)}
Trong phần này, nhóm tiến hành xây dựng các thuật toán từ đầu sử dụng Python cơ bản. Để đảm bảo triệt tiêu sai số làm tròn (round-off error), mọi phép tính phân số được bảo toàn nhờ đối tượng \code{fractions.Fraction}. Gồm các hàm cụ thể như sau:

\subsubsection{Hàm \code{gaussian\_elimination(A, b)}}
Hàm đóng vai trò nền tảng, tiến hành biến đổi ma trận $A$ và $b$ về dạng tương đương bậc thang (REF).
\begin{itemize}
    \item \textbf{Cơ chế hoạt động:} Ở mỗi cột, thuật toán tìm phần tử có trị tuyệt đối lớn nhất từ dưới lên (Partial Pivoting) rồi đưa lên vị trí chốt thông qua hoán vị dòng, bảo vệ hệ thống khỏi lỗi chia số $0$.
    \item \textbf{Cơ chế tích hợp giải nghiệm sâu:} Thuật toán tối ưu hóa mặt luồng chéo dòng bằng việc tự động trích xuất $U$, $c\_result$ để đẩy qua hàm \code{back\_substitution} nội bộ, giúp xuất khẩu ra ngoài một bộ `M, x, s` khép kín một cách hoàn hảo.
\end{itemize}
\begin{lstlisting}[language=Python, style=pythonstyle]
# Ky thuat Partial Pivoting: Tim phan tu chot lon nhat tren cot hien tai
p = r
for i in range(r + 1, n):
    if abs(M[i][c_idx]) > abs(M[p][c_idx]):
        p = i

if p != r:
    M[r], M[p] = M[p], M[r]  # Hoan vi hang de giu su on dinh
    s += 1                   # Ghi nhan so lan hoan vi cho dinh thuc
\end{lstlisting}

\subsubsection{Hàm \code{back\_substitution(U, c)}}
Tiến hành truy hồi nghiệm từ chốt cuối cùng lên trên sau khi khử về dạng bậc thang.
\begin{itemize}
    \item \textbf{Tự động quét hạng động (rank):} Thuật toán tự động quét tìm dòng chứa giá trị $\neq 0$ để định lượng chính xác rank, cho phép ứng phó với bất kỳ định dạng bậc thang nào bị khuyết phần tử.
    \item \textbf{Cấu trúc tham số linh hoạt:} Với tình huống vô số nghiệm, nhóm sử dụng \code{Dictionary} để biểu diễn các hệ số tham số tổng quát dưới dạng ánh xạ.
\end{itemize}
\begin{lstlisting}[language=Python, style=pythonstyle]
# Bieu dien cac he so tham so tong quat duoi dang anh xa Dictionary
# VD: x[j] = {'const': gia_tri, index_bien_tu_do: he_so, ...}
x_expr = [{} for _ in range(n_cols)]

for j in free_cols:
    x_expr[j] = {'const': Fraction(0), j: Fraction(1)} # Kich hoat bien tu do
\end{lstlisting}

\subsubsection{Hàm \code{inverse(A)}}
Tìm nghịch đảo ma trận bằng cách biến đổi đồng thời trên vùng không gian mở rộng.
\begin{itemize}
    \item \textbf{Cơ chế hoạt động:} Bắt đầu từ trục tọa độ $[A | I_n]$, thuật toán chuẩn hóa đồng độ cho pivot về $1$, sau đó tiến hành quét sạch (trừ khử) toàn bộ các phần tử ở DƯỚI lẫn TRÊN trục pivot, cưỡng ép phần $A$ về ma trận $I$, đẩy $A^{-1}$ sang phải.
    \item \textbf{Bắt lỗi suy biến:} Ngay tại vòng lặp, nếu bắt gặp toàn cột bằng $0$ hoặc $\det(A) = 0$, hệ thống rẽ nhánh \code{None} tức thì để ngăn chặn hiện tượng crash hệ thống.
\end{itemize}
\begin{lstlisting}[language=Python, style=pythonstyle]
# Khu ca cac dong tren lan duoi de dua ma tran truc trai ve don vi I
for i in range(n):
    if i == k: continue   
    factor = M[i][k]
    for j in range(2 * n):
        M[i][j] -= factor * M[k][j]
    M[i][k] = Fraction(0)  # Tru gian tiep luon dat sai so ve 0 tuyet doi
\end{lstlisting}

\subsubsection{Hàm \code{determinant(A)}}
Hàm này áp dụng kết quả của phép khử để tính toán $\det$ theo tốc độ đa thức $\mathcal{O}(n^3)$.
\begin{itemize}
    \item \textbf{Tối ưu không gian:} Thu được lập tức nhờ tích đường chéo chính của khối bậc thang ở trên, nhân với dấu điều hướng $(-1)^s$ từ quá trình Partial Pivoting.
\end{itemize}
\begin{lstlisting}[language=Python, style=pythonstyle]
# M[i][i] la cac chot cua ma tran bac thang, tich cua chung -> det(U)
det = Fraction((-1) ** s)  # Bat dau voi dau dieu huong
for i in range(n):
    det *= M[i][i]
\end{lstlisting}

\subsubsection{Hàm \code{rank\_and\_basis(A)}}
Dò các điểm nối nền tảng của không gian vector tạo nên ma trận ban đầu.
\begin{itemize}
    \item \textbf{Cơ sở dòng và cột:} Trích cơ sở dòng từ biến cơ sở của $U$. Tuy vậy, thuật toán phải tra ngược thông qua index list để trích xuất dữ liệu gốc của cơ sở cột ngay trên trục của ma trận gốc.
    \item \textbf{Cơ sở nghiệm:} Cài đặt tuần tự các điểm neo (biến tự do). Giữ một điểm bằng $1$, các điểm khác về $0$, giải nghiệm hệ tuyến tính sinh ra các vector trục trực giao chính xác cho toàn bộ không gian nghiệm.
\end{itemize}
\begin{lstlisting}[language=Python, style=pythonstyle]
# Giai thuat sinh Co so Khong gian nghiem (Null Space Basis, Ax = 0)
for free_j in free_cols:
    special_solution = [Fraction(0)] * n_cols
    special_solution[free_j] = Fraction(1)  # Kich hoat diem Neo thu j

    for i, k in reversed(pivot_indices):
        sum_ax = sum(U[i][j] * special_solution[j] for j in range(k + 1, n_cols))
        special_solution[k] = -sum_ax / U[i][k]
    
    null_basis.append([str(val) for val in special_solution])
\end{lstlisting}

\subsubsection{Hàm \code{verify\_solution(A, x, b)}}
Quá trình kiểm tra được chia làm hai kịch bản rõ ràng và được thẩm định bằng \code{numpy.allclose(atol=1e-9)}:
\begin{itemize}
    \item \textbf{Trường hợp nghiệm duy nhất:} Chỉ cần ép vòng mảng phân số sang \code{float} và nhân ngược lại $A \cdot x$. Nếu kết quả xấp xỉ bằng $b$, chứng tỏ thuật toán tính đúng.
    \item \textbf{Trường hợp vô số nghiệm:} Vì nghiệm trả về dưới dạng \code{Dictionary} khá phức tạp, nhóm đã phân tách nghiệm tổng quát ra làm hai phần để tự đối chiếu: 
    \begin{enumerate}
        \item \textit{Nghiệm riêng (phần hằng số $x_p$):} Kiểm tra xem $A \cdot x_p = b$ có khớp không.
        \item \textit{Nghiệm thuần nhất (các vector chứa neo biến tự do $v_k$):} Đối với mỗi tham số $t_k$, nhóm tiến hành bóc tách vector hệ số tương ứng và nhân ngược với $A$. Nếu $A \cdot v_k = \mathbf{0}$, nghĩa là vector đó thực sự khớp với không gian nghiệm.
    \end{enumerate}
\end{itemize}
\begin{lstlisting}[language=Python, style=pythonstyle]
# Doan code ma nhom dung de kiem tra chat che loi giai vo so nghiem:
# 1. Kiem tra phan hang so lam nhiem vu nghiem rieng: A * x_p = b
x_p = np.array([float(x[i].get('const', Fraction(0))) for i in range(n_cols)])
if not np.allclose(np.dot(A_np, x_p), b_np, atol=1e-9):
    print("-> SAI: Phan hang so khong khop!")
    return

# 2. Kiem tra phan vector tuong ung voi tung bien tu do: A * v_k = 0
for k in free_vars:
    v_k = np.array([float(x[i].get(k, Fraction(0))) for i in range(n_cols)])
    if not np.allclose(np.dot(A_np, v_k), np.zeros(len(b)), atol=1e-9):
        print(f"-> SAI: Vector he so cua bien tu do o t{k+1} khong thuoc Null Space!")
        return
\end{lstlisting}

\section{Phần 2: Phân Rã Ma Trận và Trực Quan Hóa với Manim}

\subsection{Chéo hóa ma trận (Matrix Diagonalization)}
\subsubsection{Cơ sở lý thuyết}
\textbf{Định nghĩa:} Một ma trận vuông $A$ cấp $n \times n$ được gọi là chéo hóa được (diagonalizable) nếu tồn tại một ma trận khả nghịch $P$ và một ma trận đường chéo $D$ sao cho:  
\begin{equation}
    A = P \cdot D \cdot P^{-1}
\end{equation}
Trong đó:
\begin{itemize}
    \item $D = \text{diag}(\lambda_1, \lambda_2, \dots, \lambda_n)$ là ma trận chứa các \textit{giá trị riêng} (eigenvalues) của $A$ trên đường chéo chính.
    \item $P = [v_1 | v_2 | \dots | v_n]$ là ma trận mà các cột là các \textit{vector riêng} (eigenvectors) tương ứng với các giá trị riêng đó.
\end{itemize}

\textbf{Điều kiện cần và đủ:} Ma trận $A$ cấp $n$ chéo hóa được khi và chỉ khi nó có $n$ vector riêng độc lập tuyến tính. Điều này tương đương với việc tổng số chiều của các không gian riêng (Geometric Multiplicity) phải bằng $n$. Nếu ma trận có $n$ giá trị riêng phân biệt thì chắc chắn chéo hóa được.

\subsubsection{Thuật toán cài đặt}
Nhóm triển khai quy trình chéo hóa (trong \code{diagonalization.py}), ma trận \(n \times n\) với \(n \le 5\) thì tự cài đặt thủ công bằng cách \textbf{Lặp QR} để rút trị riêng. Nhưng với ma trận \(n > 5\) thì nhóm sử dụng hàm \code{eigvals} của thư viện \code{numpy.linalg} để rút trị riêng. Cuối cùng sẽ khử Gauss (kế thừa từ phần 1) để xây vector riêng.

\textbf{A. Xử lý triệt để Ma trận khuyết (Defective Matrix)}\\
Điểm sáng thuật toán của nhóm là hàm \code{CheckDefective\_Eigvals}. Trước khi dựng phương trình vector riêng, thuật toán sẽ đối chiếu Bội hình học (Geometric Multiplicity - GM) và Bội đại số (Algebraic Multiplicity - AM). Nếu $GM < AM$, ma trận lập tức bị từ chối chéo hóa nhằm tránh ma trận $P$ bị suy biến.
\begin{lstlisting}[language=Python, style=pythonstyle]
# Doan code chan dung loi toan hoc the hien tu duy Chat che
AM = rounded_eigs.count(lamda) # Tinh Boi dai so
if AM > 1:
    A_minus_lamdaI = subtract_lambda_I(A, lamda)
    res_temp = rank_and_basis(A_minus_lamdaI)
    rank = res_temp['rank']
    GM = n - rank  # Tinh Boi hinh hoc tu Hang (Rank-Nullity)
    
    if GM < AM:    # Phat hien ma tran khuyet, thieu he vector!
        return True
\end{lstlisting}

\textbf{B. Phân rã RQ $\rightarrow$ QR hội tụ trị riêng}\\
Áp dụng phép lặp trực giao Gram-Schmidt biến $A_k = Q \cdot R$, nhân ngược $A_{k+1} = R \cdot Q$. Cấu trúc này dồn phần tử ngoài đường chéo về 0, chừa lại các trị riêng ròng ở đường chéo trung tâm. Các chiều không gian sau đó được quét bằng \code{rank\_and\_basis}.

\subsection{Phân rã Giá trị suy biến (SVD)}
\subsubsection{Cơ sở lý thuyết}
Phân rã SVD là phương pháp phân tích tổng quát áp dụng cho mọi ma trận $A \in \mathbb{R}^{m \times n}$. Ma trận $A$ được phân tích thành:
\begin{equation}
    A = U \cdot \Sigma \cdot V^T
\end{equation}
Trong đó:
\begin{itemize}
    \item $U \in \mathbb{R}^{m \times m}$: Ma trận trực giao chứa các vector suy biến trái (các vector riêng của $AA^T$).
    \item $V \in \mathbb{R}^{n \times n}$: Ma trận trực giao chứa các vector suy biến phải (các vector riêng của $A^TA$).
    \item $\Sigma \in \mathbb{R}^{m \times n}$: Ma trận đường chéo chứa các giá trị suy biến $\sigma_1 \geq \sigma_2 \dots \geq \sigma_p \geq 0$ với $p = \min(m, n)$.
\end{itemize}
Về mặt hình học, SVD biểu diễn biến đổi tuyến tính qua 3 bước: \textit{Xoay} ($V^T$) $\rightarrow$ \textit{Co giãn trục} ($\Sigma$) $\rightarrow$ \textit{Xoay đích} ($U$).

\subsubsection{Phân rã tĩnh bằng phép Xoay Jacobi (Jacobi Eigenvalues)}
Thay vì dùng QR, nhóm lựa chọn thuật toán \textbf{Jacobi} xử lý riêng biệt cho ma trận đối xứng liên kết $M = A^T A$ vì tính ổn định số học vững chắc, mã nguồn được nhóm xây dựng tại \code{decomposition.py}.

Hàm hoạt động bằng cách đi tìm các phần tử lệch đường chéo $M[p][q]$, sau đó tính góc xoay $\theta$ để đưa nó về 0:
\begin{lstlisting}[language=Python, style=pythonstyle]
# Doan co so luong giac de triet tieu bat ky diem roi lech nao tren hang/cot
if M[p][p] == M[q][q]:
    theta = mth.pi / 4.0
else:
    theta = 0.5 * mth.atan(2.0 * M[p][q] / (M[p][p] - M[q][q]))
cos_t, sin_t = mth.cos(theta), mth.sin(theta)

# Don vi nang tuong lay goc cua M_pq tro ve duong cheo chinh
M_pp = cos_t**2 * M[p][p] - 2*sin_t*cos_t*M[p][q] + sin_t**2 * M[q][q]
M[p][q] = M[q][p] = 0.0 # Clear hoan toan khoi ban do!
\end{lstlisting}
Sau vòng lặp đưa $M[p][q]$ về 0, $M$ lộ ra ma trận đường chéo gồm các $\lambda_i$. Thuật toán kết thúc khi $\Sigma$ (bằng $\sqrt{\lambda_i}$) và dựng trục suy biến trái $U$ ($u_i = \frac{A v_i}{\sigma_i}$) độc lập hoàn toàn với thư viện.

\subsection{Các phương pháp cấu hình và giải hệ}
Nhóm đã triển khai 3 phương pháp tiếp cận chính để giải hệ phương trình $Ax = b$ trong module \code{solvers.py}:
\begin{enumerate}
    \item \textbf{Phương pháp trực tiếp - Khử Gauss (Gauss Elimination):} Cài đặt dựa trên thuật toán cốt lõi ở Phần 1 nhưng sử dụng kiểu \code{float} để mô phỏng chính xác chi phí tính toán trong môi trường thực tế, kết hợp Partial Pivoting.
    \item \textbf{Phương pháp trực tiếp - Phân rã LU:} Tách ma trận $A = LU$. Giải hệ thông qua hai bước thế tiến ($Ly=b$) và thế lùi ($Ux=y$).
    \item \textbf{Phương pháp lặp - Gauss-Seidel:} 
    Công thức lặp cho từng phần tử:
    \begin{equation}
        x_i^{(k+1)} = \frac{1}{a_{ii}} \left( b_i - \sum_{j=1}^{i-1} a_{ij} x_j^{(k+1)} - \sum_{j=i+1}^{n} a_{ij} x_j^{(k)} \right)
    \end{equation}
    Phương pháp lặp này yêu cầu điều kiện hội tụ là ma trận $A$ phải \textit{chéo trội hàng ngặt} (Strictly Diagonally Dominant): $|a_{ii}| > \sum_{j \neq i} |a_{ij}|$.
\end{enumerate}

\subsection{Đo lường thời gian thực thi (Performance Benchmark)}
Dựa trên script \code{benchmark.py}, hệ thống tự động sinh các ma trận ngẫu nhiên chéo trội (đảm bảo phương pháp lặp hội tụ) với kích thước $n \in \{50, 100, 200, 500, 1000\}$ để đo thời gian trung bình sau 5 lần chạy độc lập.

Kết quả đo lường cho thấy:
\begin{itemize}
    \item \textbf{Khử Gauss và Phân rã LU:} Thời gian tăng đáng kể khi kích thước ma trận đạt mốc 500 và 1000. Độ phức tạp tính toán thực tế bám sát đường lý thuyết $\mathcal{O}(n^3)$. Trong khi LU nhanh hơn một chút ở pha giải hệ nhiều lần, bước phân rã ban đầu vẫn là cổ chai.
    \item \textbf{Gauss-Seidel:} Thể hiện tốc độ vượt trội ở các ma trận kích thước lớn (nếu thỏa điều kiện hội tụ chéo trội), do chi phí tính toán mỗi vòng lặp chỉ xấp xỉ $\mathcal{O}(n^2)$.
\end{itemize}

\subsection{Đánh giá tính ổn định số học (Condition Number)}
Để đánh giá tác động của sai số dấu phẩy động và số điều kiện $\kappa(A)$ lên chất lượng nghiệm, nhóm thực hiện đo đạc sai số tương đối $\frac{||A\hat{x} - b||_2}{||b||_2}$:

\begin{itemize}
    \item \textbf{Ma trận xác định dương (SPD):} Đây là ma trận "Well-conditioned" (số điều kiện nhỏ). Cả 3 phương pháp Gauss, LU và Gauss-Seidel đều cho sai số tương đối rất nhỏ (cỡ $10^{-14}$ đến $10^{-16}$).
    \item \textbf{Ma trận Hilbert ($H_{ij} = \frac{1}{i+j+1}$):} Đây là ma trận nổi tiếng về sự "Ill-conditioned" (số điều kiện tăng theo hàm mũ). Dù chỉ với kích thước nhỏ $n=10$, số điều kiện đã lên tới $\approx 10^{13}$. Thực nghiệm cho thấy:
    \begin{itemize}
        \item Phương pháp Gauss và LU: Bị mất mát chữ số có nghĩa (Loss of significance), sai số tăng vọt lên mức không thể chấp nhận được do nhiễu số học.
        \item Phương pháp Gauss-Seidel: Không hội tụ do ma trận Hilbert không thỏa mãn tính chất chéo trội hàng.
    \end{itemize}
\end{itemize}
Kết luận rút ra: Khi giải các bài toán thực tế, không chỉ quan tâm đến tốc độ $\mathcal{O}(n^3)$ mà việc kiểm tra số điều kiện của hệ $\kappa(A)$ là bắt buộc để đảm bảo tính khả tín của nghiệm số.

\subsection{Tóm tắt kết quả đạt được}
Nhóm đã bám sát đề bài và hoàn thành xuất sắc các yêu cầu đề ra, cụ thể:
\begin{itemize}
    \item \textbf{Cài đặt thuật toán cốt lõi:} Triển khai hoàn toàn thủ công các thuật toán: Khử Gauss có chọn phần tử chốt (Partial Pivoting), các thuật toán phân rã như SVD bằng ngôn ngữ Python. Thuật toán đảm bảo độ chính xác nhờ cấu trúc dữ liệu \code{Fraction}, loại bỏ sai số làm tròn.
    \item \textbf{Giải quyết bài toán thực tế:} Áp dụng thành công phân rã SVD vào kỹ thuật xấp xỉ ma trận hạng thấp (Low-Rank Approximation) để tiến hành nén ảnh kỹ thuật số, hiểu hơn về tầm quan trọng của nén ảnh trong thực tế.
    \item \textbf{Trực quan hóa sinh động:} Xây dựng thành công chuỗi video học thuật bằng công cụ đồ họa \textbf{Manim}, biến các công thức đại số trừu tượng thành những đoạn phim trực tiếp giải thích cặn kẽ bản chất hình học của từng bước phân rã SVD và quá trình chéo hóa.
    \item \textbf{Phân tích độ ổn định số:} Đánh giá thực nghiệm sâu sắc về hiện tượng mất ổn định (Ill-conditioned) trên ma trận Hilbert và sự khác biệt về tốc độ hội tụ (Speed Benchmark) giữa các giải thuật.
\end{itemize}

\subsection{Khó khăn gặp phải}
Trong hành trình thực hiện đồ án, nhóm đã đối mặt và từng bước giải quyết một số rào cản nhất định:
\begin{itemize}
    \item \textbf{Tính toán khối lượng lớn:} Việc kiên quyết cài đặt bằng Python thuần (không dùng thư viện cấp C như NumPy) đã khiến thời gian thực thi thuật toán $O(n^3)$ trên các ma trận có kích thước khổng lồ ($n \ge 1000$) tăng cao, đòi hỏi phải tái cấu trúc vòng lặp nhiều lần để giảm chi phí hằng số.
    \item \textbf{Làm chủ công cụ đồ họa Manim:} Việc phải lập trình bằng code kịch bản đòi hỏi nhiều tính toán canh chỉnh tọa độ cho các đối tượng toán học để không bị vỡ bố cục, cùng với đó là việc đồng bộ nội dung, lên kịch bản và tốn tương đối nhiều thời gian để render video thành phẩm.
    \item \textbf{Xử lý ngoại lệ trong tính toán số:} Khó khăn trong việc bắt các lỗi suy biến hệ thống khi xử lý ma trận khuyết (Defective Matrix), cũng như mất mát chữ số phần thực (Loss of Significance) khi thử nghiệm giải hệ với ma trận cực kỳ xấu như ma trận Hilbert.
\end{itemize}

\subsection{Bài học rút ra}
Qua khoảng thời gian làm việc, những giá trị cốt lõi mà nhóm đã tích lũy được có thể rút kết thành:
\begin{itemize}
    \item \textbf{Giá trị của sự tối ưu phần cứng:} Hiểu rõ sự chênh lệnh to lớn về mặt hiệu năng khi chuyển đổi các vòng lặp cơ bản.
    \item \textbf{Tư duy toán học và lập trình hướng số liệu:} Nắm vững định nghĩa "Không gian vector" và các phép biến đổi điểm/trục cơ bản, từ đó mới có thể chuyển đổi thành công tư duy phân rã phương trình thành dạng hàm thuật toán có thể giải quyết bài toán lớn hơn của thực tiễn.
    \item \textbf{Kỹ năng làm việc nhóm:} Quá trình làm việc theo các thành phần chức năng (Solver, Demcomposition, Visualization) đã giúp các thành viên củng cố kỷ luật thiết kế phần mềm và rèn luyện kỹ năng giải quyết vấn đề rủi ro chung dưới áp lực của thời hạn yêu cầu.
\end{itemize}
